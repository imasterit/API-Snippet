/**
 * Shows dashboard to user
 * @param fsm
 * @param from
 * @param to
 * @param transition
 * @param data
 * @returns {Promise<boolean>}
 */
const showDashboard = async ({fsm, from, to, transition}, data) => {
    console.info(`showDashboard ${transition}:${from}=>${to} ${fsm.state}`);
    if (transition === 'goto' && from !== 'user_question' && data !== 'Warning') return;

    if (transition === 'goBack' &&
        (from === 'signingIn' || from === 'requestPassword'
            || from === 'passwordEmail' || from === 'user_registration' || from === 'language_choosing')) return false;

    const dashboardMenu = {};

    await deleteLastMessagesFromTelegram(fsm.user);

    if (fsm.user.roles.teacher) {
        let availability;
        if(fsm.user.busy) {
            availability = "⛔️You are in DND mode⛔️";
        } else {
            availability = "✅️You are Online✅️";
        }
        dashboardMenu[availability] = "/switchTeacherAvailability";

        const sendSampleQuestion = 'DEBUG - send a sample question';
        dashboardMenu[sendSampleQuestion] = "/sendSampleQuestion";
        const emptyQuestionsQueue = 'DEBUG - empty questions queue';
        dashboardMenu[emptyQuestionsQueue] = "/emptyQuestionsQueue";

        const mySubscriptions = 'My subscriptions';
        dashboardMenu[mySubscriptions] = "/mySubscriptions";
        // const getTheMoney = 'Get the money';
        // dashboardMenu[getTheMoney] = "/underConstruction";
    }

    if (fsm.user.roles.student) {
        const iNeedHelp = 'I need help';
        dashboardMenu[iNeedHelp] = "/iNeedHelp";

        const rechargeBalance = 'Add +1000 to balance';
        dashboardMenu[rechargeBalance] = "/rechargeBalance";

        const favoriteTeachers = 'My teachers';
        dashboardMenu[favoriteTeachers] = "/showFavoriteTeachers";
    }

    const userProfile = 'My profile';
    dashboardMenu[userProfile] = "/userProfile";
    // const sessionsHistory = 'My history';
    // dashboardMenu[sessionsHistory] = "/showQuestionsHistory";

    // const resendVerification = 'Resend verification';
    // dashboardMenu[resendVerification] = "/reverification";
    //
    // const resendVerification = 'DEBUG setTheClassOver';
    // const tmpdata = JSON.stringify({classId:123,students:[{studentId:'n@t.com',spentMinutes:'3'} ]});
    // dashboardMenu[resendVerification] = `/setTheClassOver`;

    const logoutText = 'Log out';
    dashboardMenu[logoutText] = "/logout";

    console.info("showDashboard showDashboardMenu:",
        dashboardMenu);

    await fsm.user.sendMenu("Dashboard", dashboardMenu, namespace);

    return dashboardMenu;
};
